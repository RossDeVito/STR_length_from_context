# GPU Python deps for the Caduceus-Ph stack. Installed by setup_gpu_env.sh.
#
# The mamba-ssm / causal-conv1d CUDA kernels are NOT listed here: setup_gpu_env.sh
# BUILDS them from source against the torch below. Building locally makes the
# kernels' C++ ABI match your torch automatically (the prebuilt wheels do not
# offer a matching abiFALSE + torch-2.5 + cpp_functions combo) and enables
# Mamba's fused fast path.
#
# Assumes Python 3.11 and a CUDA 12.x toolkit (nvcc) available for the source
# build. Install order matters (torch first, then the kernels) — that's why the
# kernels live in the script, not in this file.

# --- PyTorch (CUDA 12.1 build; runs on TSCC A100, driver 580 supports up to CUDA 13) ---
--extra-index-url https://download.pytorch.org/whl/cu121
torch==2.5.1+cu121

# --- transformers 4.x ---
# The Caduceus trust_remote_code modeling/tokenizer targets ~4.38 and isn't
# updated for the transformers 5.x major release; 4.44.2 also supports numpy 2.x.
transformers==4.44.2

# --- Everything else (numpy, pandas, lightning, captum, ...) ---
-r requirements.txt
